import "ast"
table constraint {
  head: inRef(record.parentID),
  tail: inRef(record.parentID),
  constraintComparison: inRef(record.parentID)
}
table record {
  ident: inRef(ident.parentID)
}
table ident {
  text
}
varInComparison{head: H, tail: T} :-
  constraint {
    head {
      ident {
        text: H
      }
    },
    tail {
      ident {
        text: T
      }
    }
  }.
----
text/plain
varInComparison{head: H, tail: T} :- 
  constraint{id: VconstraintID} &
  record{id: VrecordID, parentID: VconstraintID} &
  ident{parentID: VrecordID, text: H} &
  record{id: VrecordID, parentID: VconstraintID} &
  ident{parentID: VrecordID, text: T}
