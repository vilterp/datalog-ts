import "ast"

table constraint {
  id,
  record: inRef(ast.record:parentID)
}
table ast.record {
  ident: inRef(ast.ident:parentID)
}

def varInComparison{scope: I, name: N, span: S} {
  constraint{
    record {
      ident {
        text: I
      }
    },
    constraintComparison {
      scalarExpr {
        scalarTerm {
          term {
            var{text: N, span: S}
          }
        }
      }
    }
  }
}
